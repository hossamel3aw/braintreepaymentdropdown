    <script>
    let braintreeInstance;
    const paymentAmount = "1000.00";
    const allowedOrigin = "https://app.flutterflow.io/project/socialmedia-bpdi5c"; // CHANGE THIS

    // Initialize Braintree
    async function initBraintree() {
        try {
            document.getElementById('submit-payment').disabled = true;
            document.querySelector('.loading').style.display = 'block';

            // Get client token
            const response = await fetch('https://guwzgnjdnchuvypsmzue.supabase.co/functions/v1/generate-client-token', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                }
            });
            
            const { clientToken } = await response.json();

            // Create Braintree instance
            braintreeInstance = await braintree.dropin.create({
                authorization: clientToken,
                container: '#dropin-container',
                paymentOptionPriority: ['card', 'paypal'],
                card: {
                    cardholderName: true
                },
                paypal: {
                    flow: 'checkout',
                    amount: paymentAmount,
                    currency: 'USD'
                }
            });

            document.getElementById('payment-modal').style.display = 'flex';
            document.getElementById('submit-payment').disabled = false;
            document.querySelector('.loading').style.display = 'none';
            
        } catch (error) {
            console.error('Initialization Error:', error);
            window.parent.postMessage({ 
                type: 'paymentError',
                message: 'Payment system initialization failed'
            }, allowedOrigin);
            closePayment();
        }
    }

    // Handle Payment Submission
    async function processPayment() {
        try {
            document.getElementById('submit-payment').disabled = true;
            document.querySelector('.loading').style.display = 'block';

            const payload = await braintreeInstance.requestPaymentMethod();
            
            // Process payment
            const result = await fetch('https://guwzgnjdnchuvypsmzue.supabase.co/functions/v1/process-payment', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    nonce: payload.nonce,
                    amount: paymentAmount,
                    currency: 'USD',
                    deviceData: payload.deviceData
                })
            });

            const response = await result.json();
            
            if(response.success) {
                window.parent.postMessage({
                    type: 'paymentSuccess',
                    transactionId: response.id,
                    amount: paymentAmount,
                    currency: 'USD'
                }, allowedOrigin);
            } else {
                throw new Error(response.error || 'Payment processing failed');
            }
            
            closePayment();
            
        } catch (error) {
            window.parent.postMessage({
                type: 'paymentError',
                message: error.message
            }, allowedOrigin);
            document.getElementById('submit-payment').disabled = false;
            document.querySelector('.loading').style.display = 'none';
        }
    }

    function closePayment() {
        document.getElementById('payment-modal').style.display = 'none';
        if(braintreeInstance) {
            braintreeInstance.teardown();
        }
        document.querySelector('.loading').style.display = 'none';
        document.getElementById('submit-payment').disabled = false;
    }

    // Event Listeners
    document.getElementById('submit-payment').addEventListener('click', processPayment);
    
    // Expose to FlutterFlow
    window.showPaymentUI = initBraintree;
    </script>
